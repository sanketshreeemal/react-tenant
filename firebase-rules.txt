// Firestore Rules
// Copy these rules to your Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write access to all users for all documents
    match /{document=**} {
      allow read, write: if true;
    }
    
    // For production, you should restrict access based on authentication
    // Example of more secure rules:
    // match /tenants/{tenantId} {
    //   allow read: if request.auth != null;
    //   allow write: if request.auth != null;
    // }
  }
}

// Storage Rules
// Copy these rules to your Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow read/write access to all users for all files
    match /{allPaths=**} {
      allow read, write: if true;
    }
    
    // For production, you should restrict access based on authentication
    // Example of more secure rules:
    // match /tenants/{userId}/{allPaths=**} {
    //   allow read: if request.auth != null;
    //   allow write: if request.auth != null;
    // }
  }
}
